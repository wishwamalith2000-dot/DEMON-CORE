import React, { useState, useEffect, useRef } from 'react';
import { motion } from 'framer-motion';

export const Terminal = () => {
    const [lines, setLines] = useState<string[]>([
        "SPECTRE v13.1 SHELL [SECURE]",
        "Connected to Localhost:3000",
        "Ready for injection...",
        " "
    ]);
    const [input, setInput] = useState("");
    const endRef = useRef<null | HTMLDivElement>(null);

    const handleCommand = (e: React.KeyboardEvent) => {
        if (e.key === 'Enter') {
            const newLines = [...lines];
            newLines.push(`> ${input}`);
            
            // Simulation of processing
            if (input.includes('status')) {
                newLines.push("SYSTEM STATUS: OPTIMAL");
                newLines.push("NODES: 3/3 ONLINE");
            } else if (input.includes('generate')) {
                newLines.push("ROUTING TO TIER: HIGH...");
                newLines.push("VALIDATING INTENT...");
                newLines.push("OUTPUT GENERATED [200ms]");
            } else {
                newLines.push(`Command '${input.split(' ')[0]}' recognized. Executing in SIMULATION mode.`);
            }
            
            setLines(newLines);
            setInput("");
        }
    };

    useEffect(() => {
        endRef.current?.scrollIntoView({ behavior: 'smooth' });
    }, [lines]);

    return (
        <div className="h-full w-full bg-black font-mono text-sm p-6 overflow-hidden flex flex-col hud-border">
            <div className="flex-1 overflow-y-auto space-y-1 pb-4">
                {lines.map((line, i) => (
                    <div key={i} className={`${line.startsWith('>') ? 'text-white' : 'text-spectre-green opacity-80'}`}>
                        {line}
                    </div>
                ))}
                <div ref={endRef} />
            </div>
            <div className="h-12 border-t border-gray-800 flex items-center gap-2">
                <span className="text-spectre-green">{'>'}</span>
                <input 
                    type="text" 
                    value={input}
                    onChange={(e) => setInput(e.target.value)}
                    onKeyDown={handleCommand}
                    className="bg-transparent border-none outline-none flex-1 text-white placeholder-gray-700"
                    placeholder="Enter directive..."
                    autoFocus
                />
            </div>
        </div>
    );
};
